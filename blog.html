AI-Car

Mission

Build a "car" that continues moving forward, avoid the first obstacle and stop before colliding the second obstacle.

Components

  • Adruino UNO x 1
  • Continuous Rotation Servo motors x 2
  • Portable Power Bank x 1
  • Body of the "car" x 1
  • Mini-Trolley Wheels x 1
  • Normal Wheels x 2

    Steps

    We were only given a Adruino UNO and a Mini-Trolley Wheel, thus we must make the car body and the wheels using fusion 360

    Step 1(Prototype)

    We came up with a simple design, which is a flat board with stoppers at the top surrounding the portable power bank and the audrino to prevent them from falling off when the "car" is moving. We used cardboards to make a simple prototype which allows us to visualise better and to get the dimensions.

    Step 2(Laser Cut)

    We used fusion 360 to make the components.(link to the fusion 360 file).

    Then we proceed with laser cutting.

    Step 3(assembly)

    We tried to fit the trolley wheel to the board,however it did not fit the board hence we have to drill a hole so it cana fit the baord.We glued rubber band to the wheel to increase the friction of the wheel .We also used superglute to stick the component to the board

    Step 4(coding)

    We came up with this code for our car.

    #include "Servo.h"

    #define trigpin 6 //connect trigo pin of the Ultra-Sound sensor to pin 6

    #define echopin 2 //connect echo pin of the Ultra-Sound sensor to pin 2

    Servo Lservo; //left servo

    Servo Rservo; //right servo

    const int Lstopangle=85;

    const int Rstopangle=95;

    int count=0; //obstacle count

    void setup(){

    Lservo.attach(10); //left servo connect to pin 10

    Rservo.attach(9); //right servo connect to pin 9

    pinMode(trigpin,OUTPUT);

    pinMode(echopin,INPUT);

    Serial.begin(9600 );

    }

    void loop(){

    digitalWrite(trigpin,LOW);

    delayMicroseconds(10);

    digitalWrite(trigpin,HIGH);

    delayMicroseconds(10);

    int dist=pulseIn(echopin,HIGH,8000)/58;

    Serial.println(dist);

    if (dist<=20 && dist!=0) //when the distance between the sound sensor and the obstacle is less than or equals to 20

    {

    if (count>=1){ //if this is the second obstacle, disconnect both servo

    Lservo.detach();

    Rservo.detach();

    }

    //if this the first obstacle,

    turn_left(); //turn left for 0.5 seconds,

    delay(500);

    carFwdDelay(3000); //move forward for 3 seconds,

    turn_right(); //turn right for 0.5 seconds

    delay(500);

    count++; //increase obstacle count by 1

    }

    else

    {

    carFwd(); //if no obstacles detected, continue moving forawrd

    }

    }

    void carFwd()

    {

    Lservo.write(0);

    Rservo.write(180);

    }

    void carFwdDelay(int t)

    {

    Lservo.write(0);

    Rservo.write(180);

    delay(t);

    }

    void turn_left()

    {

    Lservo.write(0);

    Rservo.write(0);

    }

    void turn_right()

    {

    Lservo.write(180);

    Rservo.write(180);

    }
